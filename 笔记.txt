{
	int beg=trans(x,y);
	q.push(beg);
	memset(d,-1,sizeof (d));
	d[x][y]=0;
		while (	!q.empty())
		{
			int t = q.front();
			q.pop();
			x=t/10000;
			y=t%10000;
			vis[x][y]=1;
			if(d[x-1][y]==0)
			if(judge(x-1,y)) Q.push(trans(x-1,y));

}
